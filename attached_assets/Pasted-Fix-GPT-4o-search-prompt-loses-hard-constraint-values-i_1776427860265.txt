Fix: GPT-4o search prompt loses hard constraint values — inject them as natural-language instructions
The mission extractor correctly produces structured constraints with real values (e.g. name contains "Swan", relationship with "local authority"). But before those reach the GPT-4o search prompt, they get stripped to generic type labels like name_contains or relationship_check. As a result, the GPT-4o prompt says "They must: name_contains." instead of "The business name must contain the word 'Swan'."
GPT-4o only learns the actual constraint value if the intent narrative's entity_description happens to include it — which is a soft, fragile channel. On harder queries (relationships, website evidence, attribute checks) this causes search to be under-specified.
The gp_cascade path is unaffected because it reads c.value directly for client-side filtering. This fix only touches the GPT-4o search path.

It's already passed into executeGpt4oPrimaryPath via ctx.structuredConstraints
It's just typed as Record<string, unknown>[] in Gpt4oSearchContext, which loses the type safety and isn't used in the prompt

2. Add a formatConstraintForSearch helper
Add this helper function near the top of the file, above buildSearchPrompt. It turns a structured constraint into a single natural-language instruction that a human (and GPT-4o) can read and act on.
ts/**
 * Convert a structured constraint into a natural-language instruction
 * suitable for a web search prompt.
 *
 * The goal is that the real constraint VALUE (e.g. "swan", "local authority",
 * "vegan") reaches GPT-4o as a clear directive, not hidden inside a narrative
 * phrase or stripped to a type label.
 */
function formatConstraintForSearch(c: StructuredConstraintPayload): string {
  const valueRaw = c.value === null || c.value === undefined ? '' : String(c.value).trim();
  if (!valueRaw) return '';

  // Capitalise first letter of word values for readability in prompts
  const valuePretty = valueRaw.length > 0
    ? valueRaw[0].toUpperCase() + valueRaw.slice(1)
    : valueRaw;

  switch (c.type) {
    case 'text_compare': {
      const fieldLabel = c.field === 'name' ? 'business name' : c.field;
      switch (c.operator) {
        case 'contains':
          return `The ${fieldLabel} must contain the word "${valuePretty}".`;
        case 'starts_with':
          return `The ${fieldLabel} must start with "${valuePretty}".`;
        case 'ends_with':
          return `The ${fieldLabel} must end with "${valuePretty}".`;
        case 'equals':
          return `The ${fieldLabel} must be exactly "${valuePretty}".`;
        case 'not_contains':
          return `The ${fieldLabel} must NOT contain "${valuePretty}".`;
        default:
          return `The ${fieldLabel} ${c.operator} "${valuePretty}".`;
      }
    }

    case 'relationship_check':
      return `They must have evidence of ${c.operator === 'serves' ? 'working with' : c.operator} ${valueRaw}.`;

    case 'website_evidence':
      return `Their website must contain evidence of "${valueRaw}".`;

    case 'attribute_check': {
      // Booleans render as "They must be independent." etc.
      if (valueRaw === 'true' || valueRaw === 'yes') {
        return `They must be ${c.field.replace(/_/g, ' ')}.`;
      }
      if (valueRaw === 'false' || valueRaw === 'no') {
        return `They must NOT be ${c.field.replace(/_/g, ' ')}.`;
      }
      return `Their ${c.field.replace(/_/g, ' ')} must be "${valueRaw}".`;
    }

    case 'status_check':
      return `They must currently be ${valueRaw}.`;

    case 'numeric_range': {
      const fieldLabel = c.field.replace(/_/g, ' ');
      switch (c.operator) {
        case 'gte':
        case '>=':
          return `Their ${fieldLabel} must be at least ${valueRaw}.`;
        case 'lte':
        case '<=':
          return `Their ${fieldLabel} must be at most ${valueRaw}.`;
        case 'gt':
        case '>':
          return `Their ${fieldLabel} must be greater than ${valueRaw}.`;
        case 'lt':
        case '<':
          return `Their ${fieldLabel} must be less than ${valueRaw}.`;
        case 'equals':
        case '=':
          return `Their ${fieldLabel} must equal ${valueRaw}.`;
        default:
          return `Their ${fieldLabel} ${c.operator} ${valueRaw}.`;
      }
    }

    case 'time_constraint':
    case 'time_predicate':
      if (c.operator === 'within_last' || c.operator === 'since') {
        return `They must have ${c.field === 'opened' ? 'opened' : c.field} ${c.operator === 'within_last' ? 'within the last' : 'since'} ${valueRaw}.`;
      }
      return `Their ${c.field} must satisfy: ${c.operator} ${valueRaw}.`;

    case 'location_constraint':
      // Location is already injected separately — skip
      return '';

    case 'ranking':
      return `Rank results by ${valueRaw}.`;

    default:
      // Safe fallback — include the real value, not just the type label
      return `They must match: ${c.field} ${c.operator} "${valueRaw}".`;
  }
}

// (not the label) so search queries reflect what the user actually asked for.
const primaryConstraintValue = (() => {
  const firstHardStructural = ctx.structuredConstraints.find(
    c => c.hardness === 'hard' && c.value !== null && c.value !== undefined && String(c.value).trim().length > 0
  );
  if (firstHardStructural) {
    const v = String(firstHardStructural.value).trim();
    // Wrap multi-word values in quotes for search engines
    return v.includes(' ') ? `"${v}"` : v;
