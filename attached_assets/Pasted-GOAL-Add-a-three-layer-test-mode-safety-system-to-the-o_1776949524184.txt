export async function sendOutreachEmail(input: OutreachSendInput): Promise<OutreachSendResult> {
  try {
    // ── LAYER 1: Test-mode redirect ──────────────────────────────────
    const testMode = (process.env.OUTREACH_TEST_MODE || '').toLowerCase() === 'true';
    let finalRecipient = input.recipientEmail;
    let finalSubject = input.subject;
    let finalBodyText = input.bodyText;
    let finalBodyHtml = input.bodyHtml;

    if (testMode) {
      const redirectTo = process.env.OUTREACH_TEST_REDIRECT_EMAIL;
      if (!redirectTo) {
        throw new Error(
          'OUTREACH_TEST_MODE=true but OUTREACH_TEST_REDIRECT_EMAIL is not set. ' +
          'Refusing to send to real recipient. Set OUTREACH_TEST_REDIRECT_EMAIL or disable test mode.'
        );
      }
      console.log(`[OUTREACH_TRANSPORT] TEST MODE ACTIVE — redirecting ${input.recipientEmail} → ${redirectTo}`);
      const originalRecipient = input.recipientEmail;
      finalRecipient = redirectTo;
      finalSubject = `[TEST → ${originalRecipient}] ${input.subject}`;
      const textBanner =
        `--- TEST MODE ---\n` +
        `This email was redirected. Original intended recipient: ${originalRecipient}\n` +
        `In production this would have been sent to that address.\n` +
        `-----------------\n\n`;
      const htmlBanner =
        `<div style="background:#fef3c7;border:1px solid #f59e0b;padding:12px;margin-bottom:16px;font-family:monospace;font-size:12px;">` +
        `<strong>--- TEST MODE ---</strong><br>` +
        `This email was redirected. Original intended recipient: <strong>${originalRecipient}</strong><br>` +
        `In production this would have been sent to that address.<br>` +
        `<strong>-----------------</strong>` +
        `</div>`;
      finalBodyText = textBanner + input.bodyText;
      finalBodyHtml = htmlBanner + input.bodyHtml;
    }

    // ── LAYER 2: Domain allow-list ──────────────────────────────────
    const allowedDomainsRaw = process.env.OUTREACH_ALLOWED_DOMAINS || '';
    const allowedDomains = allowedDomainsRaw
      .split(',')
      .map(d => d.trim().toLowerCase())
      .filter(Boolean);
    const domainGateActive = allowedDomains.length > 0;
    let domainAllowed = true;

    if (domainGateActive) {
      const recipientDomain = finalRecipient.split('@')[1]?.toLowerCase() || '';
      domainAllowed = allowedDomains.includes(recipientDomain);
      if (!domainAllowed) {
        console.warn(
          `[OUTREACH_TRANSPORT] DOMAIN GATE blocked send to ${finalRecipient} — ` +
          `domain "${recipientDomain}" not in OUTREACH_ALLOWED_DOMAINS (${allowedDomainsRaw})`
        );
        if (supabase) {
          try {
            await supabase
              .from('outreach_messages')
              .update({
                status: 'failed',
                failed_at: new Date().toISOString(),
                updated_at: new Date().toISOString(),
              })
              .eq('id', input.messageId);
          } catch (e) {
            console.error('[OUTREACH_TRANSPORT] Failed to mark message as failed after domain gate:', e);
          }
        }
        return {
          success: false,
          resendMessageId: null,
          error: 'recipient_domain_not_allowed',
        };
      }
    }

    // ── LAYER 3: Final decision log ────────────────────────────────
    console.log(
      `[OUTREACH_TRANSPORT] SEND DECISION: messageId=${input.messageId} ` +
      `original=${input.recipientEmail} final=${finalRecipient} ` +
      `test_mode=${testMode} domain_gate_active=${domainGateActive} domain_allowed=${domainAllowed}`
    );

    // ── Send via Resend ────────────────────────────────────────────
    const { client } = getResendClient();
    console.log(`[OUTREACH_TRANSPORT] Sending email: messageId=${input.messageId} to=${finalRecipient} from=${input.fromAddress}`);

    const result = await client.emails.send({
      from: input.fromAddress,
      to: finalRecipient,
      reply_to: input.replyToAddress,
      subject: finalSubject,
      html: finalBodyHtml,
      text: finalBodyText,
      headers: {
        'X-Wyshbone-Message-Id': input.messageId,
      },
    });

    const resendId = (result as any)?.data?.id ?? (result as any)?.id ?? null;

    if (resendId) {
      console.log(`[OUTREACH_TRANSPORT] Email sent successfully: messageId=${input.messageId} resendId=${resendId}`);
      if (supabase) {
        await supabase
          .from('outreach_messages')
          .update({
            resend_message_id: resendId,
            resend_status: 'sent',
            status: 'sent',
            sent_at: new Date().toISOString(),
            updated_at: new Date().toISOString(),
          })
          .eq('id', input.messageId);
      }
      return { success: true, resendMessageId: resendId, error: null };
    }

    console.warn(`[OUTREACH_TRANSPORT] Resend returned no ID for messageId=${input.messageId}`);
    return { success: false, resendMessageId: null, error: 'No message ID returned from Resend' };
  } catch (err: any) {
    const errMsg = err.message || String(err);
    console.error(`[OUTREACH_TRANSPORT] Send failed: messageId=${input.messageId} error=${errMsg}`);
    if (supabase) {
      try {
        await supabase
          .from('outreach_messages')
          .update({
            status: 'failed',
            failed_at: new Date().toISOString(),
            updated_at: new Date().toISOString(),
          })
          .eq('id', input.messageId);
      } catch (e) {
        console.error('[OUTREACH_TRANSPORT] Failed to update message status to failed:', e);
      }
    }
    return { success: false, resendMessageId: null, error: errMsg };
  }
}

Notes:
- The function signature is unchanged.
- The outreach_messages DB row is NOT updated to reflect the redirected recipient — the DB keeps the original intended recipient for audit. The redirection only affects what's sent to Resend.
- Fail-closed: if OUTREACH_TEST_MODE=true but the redirect email is missing, the function throws rather than sending to the real recipient.

Create a new file at /server/supervisor/seed-outreach-config.ts with this exact content:

/**
 * seed-outreach-config.ts
 *
 * Idempotent seed of the outreach_config row for the demo user.
 * Runs once on server startup. Safe to re-run — uses upsert on user_id.
 */

import { supabase } from '../supabase';

const DEMO_USER_ID = '8f9079b3ddf739fb0217373c92292e91';

export async function seedDemoOutreachConfig(): Promise<void> {
  if (!supabase) {
    console.warn('[SEED_OUTREACH] Supabase not configured — skipping seed');
    return;
  }

  try {
    const { error } = await supabase
      .from('outreach_config')
      .upsert(
        {
          user_id: DEMO_USER_ID,
          display_name: 'Phil',
          handle: 'phil',
          sending_domain: 'wyshbonesales.com',
          reply_to_domain: 'wyshbonesales.com',
          user_real_email: 'phil@wyshbonesales.com',
          signature_text: null,
          enabled: true,
          updated_at: new Date().toISOString(),
        },
        { onConflict: 'user_id' }
      );

    if (error) {
      console.warn(`[SEED_OUTREACH] Seed failed (non-fatal): ${error.message}`);
      return;
    }

    console.log('[SEED_OUTREACH] Config seeded for demo user — from=phil@wyshbonesales.com');
  } catch (err: any) {
    console.warn(`[SEED_OUTREACH] Seed failed (non-fatal): ${err.message}`);
  }
}
