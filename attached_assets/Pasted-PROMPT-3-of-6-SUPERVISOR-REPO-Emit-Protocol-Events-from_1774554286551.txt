/**
 * Protocol Logger — emits typed lifecycle events for the UI event protocol v1
 * These are written to agent_activities with metadata.protocol = 'v1'
 * The UI reads these via /api/afr/stream and renders them without regex matching.
 */

import { supabase } from '../supabase';

interface ProtocolLogBase {
  userId: string;
  runId: string;
  conversationId?: string;
  clientRequestId?: string;
}

async function emitProtocolEvent(params: ProtocolLogBase & {
  actionTaken: string;
  label: string;
  metadata: Record<string, unknown>;
}): Promise<void> {
  if (!supabase) return;
  try {
    const { error } = await supabase.from('agent_activities').insert({
      id: `proto_${Date.now()}_${Math.random().toString(36).substring(2, 8)}`,
      user_id: params.userId,
      timestamp: Date.now(),
      task_generated: params.label,
      action_taken: params.actionTaken,
      action_params: {},
      results: null,
      interesting_flag: 0,
      status: 'success',
      error_message: null,
      duration_ms: null,
      conversation_id: params.conversationId || null,
      run_id: params.runId,
      client_request_id: params.clientRequestId || null,
      metadata: { runType: 'tool', protocol: 'v1', actionTaken: params.actionTaken, ...params.metadata },
      created_at: Date.now(),
    });
    if (error) console.warn(`[PROTO_LOG] ${params.actionTaken} failed: ${error.message}`);
  } catch (e: any) {
    console.warn(`[PROTO_LOG] ${params.actionTaken} threw: ${e.message}`);
  }
}

export async function emitPhaseEntered(base: ProtocolLogBase & {
  phaseName: string;
  phaseIcon: string;
  phaseLabel: string;
  phaseIndex: number;
  totalPhases: number;
}): Promise<void> {
  await emitProtocolEvent({
    ...base,
    actionTaken: 'phase_entered',
    label: base.phaseLabel,
    metadata: {
      phaseName: base.phaseName,
      phaseIcon: base.phaseIcon,
      phaseLabel: base.phaseLabel,
      phaseIndex: base.phaseIndex,
      totalPhases: base.totalPhases,
    },
  });
}

export async function emitMilestoneReached(base: ProtocolLogBase & {
  milestoneKey: string;
  milestoneIcon: string;
  milestoneText: string;
  phaseName: string;
}): Promise<void> {
  await emitProtocolEvent({
    ...base,
    actionTaken: 'milestone_reached',
    label: base.milestoneText,
    metadata: {
      milestoneKey: base.milestoneKey,
      milestoneIcon: base.milestoneIcon,
      milestoneText: base.milestoneText,
      phaseName: base.phaseName,
    },
  });
}

export async function emitProgressTick(base: ProtocolLogBase & {
  tickIcon: string;
  tickText: string;
  phaseName: string;
}): Promise<void> {
  await emitProtocolEvent({
    ...base,
    actionTaken: 'progress_tick',
    label: base.tickText,
    metadata: {
      tickIcon: base.tickIcon,
      tickText: base.tickText,
      phaseName: base.phaseName,
    },
  });
}

export async function emitIntentResolved(base: ProtocolLogBase & {
  entityDescription: string;
  findability?: string;
  keyDiscriminator?: string;
  exclusions?: string[];
  scarcityExpectation?: string;
}): Promise<void> {
  await emitProtocolEvent({
    ...base,
    actionTaken: 'intent_resolved',
    label: base.entityDescription,
    metadata: {
      entityDescription: base.entityDescription,
      findability: base.findability,
      keyDiscriminator: base.keyDiscriminator,
      exclusions: base.exclusions,
      scarcityExpectation: base.scarcityExpectation,
    },
  });
}

export async function emitResultsReady(base: ProtocolLogBase & {
  resultType: string;
  resultCount: number;
}): Promise<void> {
  await emitProtocolEvent({
    ...base,
    actionTaken: 'results_ready',
    label: `${base.resultCount} results ready`,
    metadata: {
      resultType: base.resultType,
      resultCount: base.resultCount,
    },
  });
}
